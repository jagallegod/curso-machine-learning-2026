\frame{
  \frametitle{Inspiración: el cerebro}

\begin{itemize}
\item El cerebro es una red de \textbf{neuronas} interconectadas, un procesador paralelo masivo \vspace{0.3cm}

\item Cada neurona capta seńales, las procesa y, si superan un \textbf{umbral}, emite una seńal \vspace{0.3cm}

\item Las \textbf{redes neuronales artificiales} imitan esa relación entre seńales de entrada y de salida

\end{itemize}
}

\frame{
  \frametitle{Neuronas artificiales}

Una neurona con $n$ entradas se representa como:
\[
y(x) = f\!\left(\sum_{i=1}^n w_i x_i\right)
\]

\begin{itemize}
\item Cada entrada $x_i$ se pondera por un \textbf{peso} $w_i$ (su importancia) \vspace{0.3cm}

\item La \textbf{función de activación} $f(\cdot)$ decide la seńal de salida a partir de la suma ponderada

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Tres componentes de una red}

\begin{itemize}
\item[1.] \textbf{Función de activación}: transforma la suma de entradas en una salida \vspace{0.3cm}

\item[2.] \textbf{Topología}: número de neuronas, de capas, y cómo se conectan \vspace{0.3cm}

\item[3.] \textbf{Algoritmo de entrenamiento}: cómo se fijan los pesos de las conexiones

\end{itemize}
}

\frame{
  \frametitle{Topología: capas}

\begin{itemize}
\item Las neuronas se organizan en \textbf{capas}: una de \textbf{entrada}, una de \textbf{salida}, y posibles capas \textbf{ocultas} \vspace{0.3cm}

\item Cada nodo de entrada procesa una característica; el de salida produce la predicción \vspace{0.3cm}

\item La versión más simple tiene una sola capa

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCuántos nodos y capas?}

\begin{itemize}
\item Los nodos de \textbf{entrada} los fija el número de características; los de \textbf{salida}, el problema \vspace{0.3cm}

\item Los nodos \textbf{ocultos} los elige el usuario: no hay regla teórica \vspace{0.3cm}

\item Más capacidad $\rightarrow$ puede aprender relaciones más complejas... \vspace{0.3cm}

\item ...pero mayor riesgo de \textbf{sobreajuste} (de nuevo, el sesgo-varianza)

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Entrenamiento: backpropagation}

\begin{itemize}
\item Se parte de pesos \textbf{aleatorios} y se itera en ciclos (\textit{épocas}), cada uno con dos fases: \vspace{0.2cm}

\begin{itemize}
\item[1.] \textbf{Hacia adelante}: se activan las neuronas de la entrada a la salida y se produce una predicción
\item[2.] \textbf{Hacia atrás}: se compara con el valor real y el \textbf{error} se propaga para ajustar los pesos
\end{itemize}
\vspace{0.2cm}

\item El ajuste usa \textbf{descenso de gradiente}: mover los pesos en la dirección que más reduce el error

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żPor qué redes especiales para imágenes?}

\begin{itemize}
\item Una imagen pequeńa ($100\times100$) ya tiene 10.000 píxeles \vspace{0.3cm}

\item Una red \textbf{densa} necesitaría millones de pesos: no escala y sobreajusta \vspace{0.3cm}

\item Además, ignoraría la \textbf{estructura espacial}: píxeles vecinos están relacionados \vspace{0.3cm}

\item Las \textbf{CNN} aprovechan esa estructura local

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Una imagen es una matriz de números}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Cada \textbf{píxel} es un número (su intensidad) \vspace{0.2cm}
\item A color son \textbf{3 matrices} (canales rojo, verde, azul) \vspace{0.2cm}
\item Una imagen $=$ un \textbf{tensor} alto $\times$ ancho $\times$ canales
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{image_pixels.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{La convolución}

\begin{itemize}
\item Un \textbf{filtro} (pequeńa matriz de pesos) recorre la imagen y detecta un patrón local (un borde, una textura)
\end{itemize}

\begin{figure}
\includegraphics[width=0.74\textwidth]{convolution.pdf}
\end{figure}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{La convolución, en detalle}

\begin{itemize}
\item En cada posición, el filtro multiplica \textbf{punto a punto} con la ventana y \textbf{suma}: un solo número
\end{itemize}

\begin{figure}
\includegraphics[width=0.85\textwidth]{conv_detail.pdf}
\end{figure}

\begin{itemize}
\item Al deslizarse por toda la imagen, produce un \textbf{mapa de características}
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Filtros y parámetros compartidos}

\begin{itemize}
\item Cada filtro produce un \textbf{mapa de características}: dónde aparece ese patrón \vspace{0.3cm}

\item Los pesos del filtro \textbf{se aprenden} (no se diseńan a mano) \vspace{0.3cm}

\item El \textbf{mismo} filtro se usa en toda la imagen: \textbf{comparte parámetros} \vspace{0.2cm}
\begin{itemize}
\item muchísimos menos pesos que una red densa
\item detecta el patrón \textbf{esté donde esté} (invarianza a traslación)
\end{itemize}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Una jerarquía de características}

\begin{figure}
\includegraphics[width=0.95\textwidth]{feature_hierarchy.pdf}
\end{figure}

\begin{itemize}
\item Al apilar capas, la red aprende sola: de \textbf{bordes} a \textbf{partes} a \textbf{objetos} completos
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Pooling}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Resume regiones (p. ej. el \textbf{máximo} de cada vecindad $2\times2$) \vspace{0.2cm}
\item Reduce el tamańo (menos cómputo) \vspace{0.2cm}
\item Da \textbf{invarianza}: el patrón se detecta aunque se mueva un poco
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{pooling.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Arquitectura de una CNN}

\begin{figure}
\includegraphics[width=0.95\textwidth]{cnn_diagram.pdf}
\end{figure}

\begin{itemize}
\item Bloques de \textbf{convolución + pooling} extraen características; al final, capas \textbf{densas} clasifican
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Ingredientes que faltan}

\begin{itemize}
\item \textbf{ReLU} como activación: rápida y evita que los gradientes se desvanezcan \vspace{0.3cm}

\item \textbf{Softmax} en la última capa: convierte la salida en probabilidades por clase \vspace{0.3cm}

\item \textbf{Dropout}: apagar neuronas al azar durante el entrenamiento, para \textbf{regularizar} y combatir el sobreajuste

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Arquitecturas famosas y \textit{transfer learning}}

\begin{itemize}
\item Hitos: LeNet (dígitos), \textbf{AlexNet} (2012, el ``big bang'' del deep learning), VGG, ResNet \vspace{0.3cm}

\item \textbf{Transfer learning}: tomar una red ya entrenada en millones de imágenes y \textbf{reajustarla} a tu problema \vspace{0.3cm}

\item Permite obtener buenos resultados con \textbf{pocos datos} propios: hoy es la práctica habitual

\end{itemize}
}

\frame{
\begin{center}
{\Huge \textbf{Parte 3}}\\[0.5cm]
{\large Redes recurrentes (RNNs)}
\end{center}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Datos secuenciales}

\begin{itemize}
\item Muchos datos son \textbf{secuencias} donde el \textbf{orden} importa: \vspace{0.2cm}
\begin{itemize}
\item texto (una palabra depende de las anteriores)
\item series de tiempo (precios, clima)
\item audio, video
\end{itemize}
\vspace{0.3cm}

\item Una red normal trata cada entrada por separado: no tiene \textbf{memoria} del pasado

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{La idea: recurrencia y memoria}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Una \textbf{RNN} tiene conexiones que vuelven hacia atrás (\textit{loops}) \vspace{0.2cm}
\item Mantiene un \textbf{estado} que resume lo visto hasta ahora \vspace{0.2cm}
\item Así incorpora \textbf{memoria} de corto plazo
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{recurrent.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Una RNN desplegada en el tiempo}

\begin{figure}
\includegraphics[width=0.88\textwidth]{rnn_unrolled.pdf}
\end{figure}

\begin{itemize}
\item En cada paso $t$ entra $x_t$, se actualiza el estado $h_t$ (usando $h_{t-1}$) y se produce $y_t$
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCómo procesa una RNN?}

\begin{itemize}
\item En cada paso combina la entrada $x_t$ con el estado previo $h_{t-1}$:
\[
h_t = f\!\left(W\,x_t + U\,h_{t-1} + b\right)
\]

\item El estado $h_t$ \textbf{resume} todo lo visto hasta el momento (la memoria) \vspace{0.3cm}

\item Los \textbf{mismos} pesos ($W$, $U$) se reutilizan en cada paso: comparte parámetros \textbf{en el tiempo} (como el filtro de la CNN lo hace en el espacio)

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Tipos de tareas secuenciales}

\begin{figure}
\includegraphics[width=0.98\textwidth]{rnn_task_types.pdf}
\end{figure}

\begin{itemize}
\item La misma idea sirve para clasificar, traducir/etiquetar o generar secuencias
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{El gradiente que se desvanece}

\begin{itemize}
\item Para entrenar, el error se propaga muchos pasos \textbf{hacia atrás en el tiempo} \vspace{0.3cm}

\item Al multiplicarse paso tras paso, los gradientes tienden a \textbf{encogerse} (o explotar) \vspace{0.3cm}

\item Resultado: la red ``olvida'' lo lejano y le cuesta aprender \textbf{dependencias largas} \vspace{0.3cm}

\item Es el problema que motivó a las LSTM

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LSTM y GRU}

\begin{itemize}
\item Las \textbf{LSTM} ańaden una \textbf{celda de memoria} con ``compuertas'' que deciden qué recordar y qué olvidar \vspace{0.3cm}

\item Así manejan mucho mejor las dependencias de \textbf{largo plazo} \vspace{0.3cm}

\item Las \textbf{GRU} son una versión más simple, con menos compuertas \vspace{0.3cm}

\item Son el estándar cuando se usan RNN en la práctica

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LSTM: las compuertas}

\begin{figure}
\includegraphics[width=0.8\textwidth]{lstm_gates.pdf}
\end{figure}

\begin{itemize}
\item \textbf{Olvidar} (qué borrar), \textbf{entrada} (qué guardar), \textbf{salida} (qué emitir): regulan el flujo de la memoria
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Atención}

\begin{columns}
\begin{column}{0.48\textwidth}
\begin{itemize}
\item En vez de comprimir todo en un estado, la \textbf{atención} deja que cada elemento ``mire'' directamente a los relevantes \vspace{0.2cm}
\item Cada palabra \textbf{pondera} cuánto atiende a las demás
\end{itemize}
\end{column}
\begin{column}{0.52\textwidth}
\includegraphics[width=\textwidth]{attention_concept.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Transformers y modelos de lenguaje}

\begin{itemize}
\item Los \textbf{Transformers} reemplazan la recurrencia por pura \textbf{atención} \vspace{0.3cm}

\item Procesan toda la secuencia \textbf{en paralelo}: más rápido y mejor con dependencias largas \vspace{0.3cm}

\item Son la base de los grandes \textbf{modelos de lenguaje} (la ``T'' de ChatGPT) \vspace{0.3cm}

\item Su punto de entrada son los \textbf{embeddings} (sesión 5): el texto convertido en vectores

\end{itemize}
}

\begin{itemize}
\item Traducción automática, generación de texto, reconocimiento de voz \vspace{0.3cm}

\item Pronóstico de series de tiempo \vspace{0.3cm}

\item \textbf{Al taller}: clasificar reseńas de cine como positivas/negativas con la base \textbf{IMDB}, usando una capa de \textit{embedding} + una \textbf{LSTM} (en \texttt{keras})

}

\frame{
  \frametitle{żCuándo (y cuándo no) usar deep learning?}

\begin{itemize}
\item Brilla con \textbf{muchos datos} y entradas \textbf{no estructuradas} (imágenes, texto, audio, video) \vspace{0.3cm}

\item Para datos \textbf{tabulares} y muestras \textbf{pequeńas}, métodos como regresión, árboles o \textit{boosting} (XGBoost) suelen igualar o superar al DL, y son más rápidos e interpretables \vspace{0.3cm}

\item Costos del DL: mucho \textbf{cómputo}, muchos \textbf{datos}, y es una \textbf{caja negra} \vspace{0.3cm}

\item Regla práctica: empieza simple y usa DL cuando el problema y los datos lo justifiquen

\end{itemize}
}

\frame{
  \frametitle{Herramientas y cómputo}

\begin{itemize}

\item Por eso, para los talleres de CNN/RNN usaremos \textbf{Google Colab}, que ofrece GPU gratis y ya trae todo instalado \vspace{0.3cm}

\item Y aprovecharemos modelos \textbf{preentrenados} (\textit{transfer learning}) cuando se pueda

\end{itemize}
}

\frame{
  \frametitle{Resumen de la sesión}

\begin{itemize}
\item Las \textbf{ANNs} imitan neuronas: suma ponderada + activación, en capas, entrenadas con \textit{backpropagation} \vspace{0.3cm}

\item \textbf{CNNs}: convolución (parámetros compartidos) + pooling para \textbf{imágenes}; jerarquía de características \vspace{0.3cm}

\item \textbf{RNNs} (y LSTM): memoria para \textbf{secuencias}; la \textbf{atención} y los \textbf{Transformers} son el presente del área

\end{itemize}

  \frametitle{Aplicación real en Colombia: FAC y FACSAT-1}

\begin{itemize}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{FACSAT-1: los métodos}

\begin{itemize}
