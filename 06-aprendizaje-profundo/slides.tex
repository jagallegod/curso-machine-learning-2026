\frame{

\frame{
  \frametitle{¢Cµmo procesa una RNN?}

\begin{itemize}
\item En cada paso combina la entrada $x_t$ con el estado previo $h_{t-1}$:

\item Los \textbf{mismos} pesos ($W$, $U$) se reutilizan en cada paso: comparte parÃmetros \textbf{en el tiempo} (como el filtro de la CNN lo hace en el espacio)

\end{itemize}
