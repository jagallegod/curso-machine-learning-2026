%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Aprendizaje no supervisado}

\begin{itemize}
\item En el supervisado teníamos un \textit{outcome} $y$ que queríamos predecir \vspace{0.3cm}

\item En el \textbf{no supervisado} no hay $y$: solo tenemos las características \vspace{0.3cm}

\item El objetivo no es predecir, sino \textbf{describir} y \textbf{descubrir patrones} en los datos \vspace{0.3cm}

\item Ninguna variable es más importante que las demás: las miramos todas en conjunto

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Tres tareas que veremos}

\begin{itemize}
\item[1.] \textbf{Clustering}: agrupar objetos similares (segmentar) \vspace{0.3cm}

\item[2.] \textbf{Detección de anomalías}: encontrar los casos ``raros'' \vspace{0.3cm}

\item[3.] \textbf{Representación / reducción de dimensión}: resumir muchas variables en pocas, más útiles \vspace{0.3cm}

\item Como no hay etiquetas, evaluar el resultado es más difícil: mucho depende de la \textbf{interpretación} del analista

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żPara qué sirve?}

\begin{itemize}
\item \textbf{Segmentación} de clientes, votantes o mercados \vspace{0.3cm}

\item \textbf{Detección de fraude}, fallas o intrusiones (lo raro) \vspace{0.3cm}

\item \textbf{Visualización} y \textbf{compresión} de datos de alta dimensión \vspace{0.3cm}

\item \textbf{Preprocesamiento}: crear mejores variables para un modelo supervisado posterior

\end{itemize}
}

\frame{
  \frametitle{Clustering}

\begin{itemize}
\item A veces conviene \textbf{agrupar} objetos en categorías que no conocíamos de antemano \vspace{0.3cm}

\item En marketing: segmentar mercados para una estrategia por grupo \vspace{0.3cm}

\item En campańas políticas: identificar tipos de votantes \vspace{0.3cm}

\item El clustering divide los datos en \textbf{clusters}: grupos de objetos similares entre sí y distintos de los demás

\end{itemize}
}

\frame{
  \frametitle{Clustering: clasificación no supervisada}

\begin{itemize}
\item Es \textbf{no supervisado}: no decimos ex ante cómo deben verse los grupos \vspace{0.3cm}

\item No sirve para \textbf{predecir}, sino para \textbf{descubrir} estructura \vspace{0.3cm}

\item Los grupos que devuelve el computador \textbf{no tienen significado explícito}: la interpretación la pone el analista \vspace{0.3cm}

\item żCómo decide el computador dónde están las fronteras de los grupos?

\end{itemize}
}

\frame{
  \frametitle{El algoritmo de k-medias}

\begin{itemize}
\item El más usado para clustering; emparentado con kNN \vspace{0.3cm}

\item Asigna cada uno de los $n$ ejemplos a uno de $k$ clusters \vspace{0.3cm}

\item El número $k$ se fija \textbf{ex ante} (de forma heurística) \vspace{0.3cm}

\item Busca \textbf{minimizar} las diferencias dentro de cada cluster y \textbf{maximizar} las diferencias entre clusters

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{k-medias: elegir centros}

\begin{itemize}
\item Se eligen al azar $k$ puntos como centros iniciales (aquí $k=3$):
\end{itemize}

\begin{figure}
\includegraphics[width=0.6\textwidth]{clusters3.pdf}
\end{figure}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{k-medias: asignar al centro más cercano}

\begin{itemize}
\item Cada punto se asigna al centro más cercano (distancia euclídea): se forma el \textbf{diagrama de Voronoi}
\[
dist(x,y)=\sqrt{\sum_{i=1}^n (x_i-y_i)^2}
\]
\end{itemize}

\begin{figure}
\includegraphics[width=0.55\textwidth]{clusters4.pdf}
\end{figure}
}

\frame{
  \frametitle{k-medias: actualizar los centroides}

\begin{itemize}
\item Cada centro se mueve al \textbf{centroide} (la posición promedio) de los puntos de su cluster:
\end{itemize}

\begin{figure}
\includegraphics[width=0.6\textwidth]{clusters5.pdf}
\end{figure}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{k-medias: iterar hasta converger}

\begin{itemize}
\item Con los nuevos centros se reasignan los puntos; algunos cambian de cluster \vspace{0.2cm}
\item Se repite hasta que ya no haya cambios:
\end{itemize}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCuántos clusters?}

\begin{itemize}
\item $k$ es una decisión crucial (y el algoritmo también es sensible a los centros iniciales) \vspace{0.3cm}

\item Si hay un \textbf{prior}, úsalo (categorías de negocio, número de mesas, de campańas...) \vspace{0.3cm}

\item Regla de dedo: $k = \sqrt{n/2}$ \vspace{0.3cm}

\item \textbf{Método del codo}: aumentar $k$ hasta que la ganancia marginal en homogeneidad se aplane

\end{itemize}
}

{\large Detección de anomalías}
\end{center}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żQué es una anomalía?}

\begin{itemize}
\item Una \textbf{anomalía} (\textit{outlier}) es un caso que se aparta mucho del patrón general \vspace{0.3cm}

\item A menudo lo raro es lo más interesante:
\begin{itemize}
\item transacciones \textbf{fraudulentas} con tarjeta
\item \textbf{fallas} de una máquina o un sensor
\item \textbf{intrusiones} en una red; errores en los datos
\end{itemize}
\vspace{0.3cm}

\item Suele ser un problema \textbf{no supervisado}: no tenemos ejemplos etiquetados de ``anomalía''

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Cuatro familias de métodos}

\begin{itemize}
\item \textbf{Estadístico}: lo que cae lejos de lo típico (z-score, rango intercuartílico) \vspace{0.3cm}

\item \textbf{Por distancia}: puntos lejos de sus vecinos o del centro de su cluster \vspace{0.3cm}

\item \textbf{Por densidad}: puntos en zonas poco pobladas (LOF) \vspace{0.3cm}

\item \textbf{Por modelo}: aprender la ``forma normal'' y marcar lo que no encaja (\textit{isolation forest}, \textit{one-class SVM})

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Enfoque estadístico}

\begin{columns}
\begin{column}{0.5\textwidth}
\footnotesize
\begin{itemize}
\item Marcar como atípico lo que cae lejos de lo \textbf{típico} \vspace{0.25cm}
\item Univariado: valores con $|z|>3$ (regla de 3 sigma) o fuera del rango intercuartílico (Tukey) \vspace{0.25cm}
\item Simple y rápido, pero mira \textbf{una variable a la vez} y asume una forma (p.\,ej. normal)
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{zscore_outliers.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Enfoque por distancia}

\begin{itemize}
\item Idea: un punto es anómalo si está \textbf{lejos} de los demás \vspace{0.3cm}

\item Distancia al $k$-ésimo vecino más cercano: a mayor distancia, más aislado \vspace{0.3cm}

\item O distancia al \textbf{centroide} de su cluster (conecta con k-medias): los puntos lejos de todos los centros son sospechosos \vspace{0.3cm}

\item Limitación: falla cuando hay zonas con densidades muy distintas

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LOF: Local Outlier Factor}

\begin{itemize}
\item Idea: comparar la \textbf{densidad local} de un punto con la de sus vecinos \vspace{0.3cm}

\item Si un punto está en una zona \textbf{mucho menos densa} que sus vecinos, es sospechoso \vspace{0.3cm}

\item Devuelve un \textbf{puntaje}:
\begin{itemize}
\item $LOF \approx 1$: densidad parecida a la de sus vecinos (normal)
\item $LOF \gg 1$: mucho más aislado que sus vecinos (anomalía)
\end{itemize}
\vspace{0.3cm}

\item Ventaja sobre el enfoque por distancia: detecta anomalías \textbf{locales}, no solo las globales

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LOF: żcómo se calcula? (intuición)}

\begin{itemize}
\item[1.] Para cada punto se miran sus $k$ vecinos más cercanos \vspace{0.15cm}

\item[2.] Su \textbf{densidad local} se aproxima como el inverso de la distancia promedio a esos vecinos:
\[
\text{densidad}(p)\;\approx\;\frac{1}{\text{dist. promedio de } p \text{ a sus } k \text{ vecinos}}
\]

\item[3.] El \textbf{LOF} compara esa densidad con la de sus vecinos:
\[
LOF(p)\;\approx\;\frac{\text{densidad promedio de los vecinos de } p}{\text{densidad de } p}
\]

\item[4.] $LOF\approx 1$: normal; $\;LOF\gg 1$: zona mucho menos densa que la de sus vecinos $\Rightarrow$ anomalía

\end{itemize}
}

\frame{
  \frametitle{LOF en acción}

\begin{figure}
\includegraphics[width=0.6\textwidth]{lof_anomalias.pdf}
\end{figure}

\begin{center}
\small Los puntos con LOF alto (rojos) están en zonas poco densas, lejos de los grupos.
\end{center}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Isolation Forest}

\begin{itemize}
\item Enfoque \textbf{por modelo}, basado en árboles (conecta con random forests) \vspace{0.25cm}

\item Construye árboles con \textbf{cortes aleatorios}: elige una variable al azar y un punto de corte al azar, y repite \vspace{0.25cm}

\item Cada corte parte los datos en dos; se sigue hasta \textbf{aislar} cada punto \vspace{0.25cm}

\item Un punto \textbf{anómalo} se aísla con \textbf{pocos cortes} (está apartado); uno \textbf{normal}, en zona densa, requiere \textbf{muchos} \vspace{0.25cm}

\item La anomalía se mide por el número \textbf{promedio} de cortes (pocos = anómalo); rápido y escala bien

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Isolation Forest: la intuición}

\begin{columns}
\begin{column}{0.46\textwidth}
\footnotesize
\begin{itemize}
\item Cada \textbf{nodo} del árbol es un corte aleatorio \vspace{0.3cm}
\item La \textbf{anomalía} (rojo) cae en una hoja \textbf{poco profunda}: pocos cortes la aíslan \vspace{0.3cm}
\item Un punto \textbf{normal} (azul) cae en una hoja \textbf{profunda}: hacen falta muchos cortes \vspace{0.3cm}
\item Puntaje = profundidad \textbf{promedio} en muchos árboles
\end{itemize}
\end{column}
\begin{column}{0.54\textwidth}
\includegraphics[width=\textwidth]{isoforest.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCómo evaluar sin etiquetas?}

\begin{itemize}
\item Sin etiquetas de ``anomalía'', validar es difícil: a menudo se \textbf{inspeccionan a mano} los casos con mayor puntaje \vspace{0.3cm}

\item Si tenemos \textbf{algunas} etiquetas (p. ej. fraudes confirmados), es un problema \textbf{semi-supervisado}: podemos medir cuántos de los \textit{top-N} marcados eran reales (precisión@N) \vspace{0.3cm}

\item No hay umbral universal: cuántos casos marcar depende del \textbf{costo} de cada error \vspace{0.3cm}

\item Como se basa en distancias, conviene \textbf{escalar} las variables (como en kNN)

\end{itemize}
}

\frame{
\begin{center}
{\Huge \textbf{Parte 4}}\\[0.5cm]
{\large Representación de datos y \textit{embeddings}}
\end{center}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{El reto de la alta dimensión}

\begin{itemize}
\item Muchos problemas tienen \textbf{decenas o miles} de variables \vspace{0.3cm}

\item Difíciles de \textbf{visualizar}, costosas de procesar, y con la \textbf{maldición de la dimensionalidad} \vspace{0.3cm}

\item Pero muchas variables están \textbf{correlacionadas}: hay redundancia \vspace{0.3cm}

\item żPodemos \textbf{resumir} la información en muchas menos dimensiones, perdiendo poco?

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Extracción de características}

\begin{itemize}
\item La \textbf{selección} de variables elige un subconjunto; la \textbf{extracción} crea variables \textbf{nuevas y compuestas} \vspace{0.3cm}

\item Es la versión no supervisada del \textit{feature extraction} de la sesión 4 \vspace{0.3cm}

\item La técnica clásica: \textbf{Análisis de Componentes Principales} (PCA), que resume muchas variables \textbf{correlacionadas} en pocas \vspace{0.3cm}

\item La matemática exacta se sale del curso; vamos por la \textbf{intuición}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{PCA: la intuición}

\begin{columns}
\begin{column}{0.46\textwidth}
\begin{itemize}
\item Para predecir si alguien paga un crédito hay decenas de atributos: edad, educación, ingreso, ocupación, código postal\ldots \vspace{0.2cm}
\item Muchos están \textbf{correlacionados}: reflejan pocas \textbf{dimensiones ocultas} ---p.\,ej. ``responsabilidad'' y ``afluencia'' \vspace{0.2cm}
\item PCA \textbf{descubre} esas dimensiones compuestas
\end{itemize}
\end{column}
\begin{column}{0.54\textwidth}
\includegraphics[width=\textwidth]{loan_concept.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żQué hace PCA?}

\begin{columns}
\begin{column}{0.52\textwidth}
\begin{itemize}
\item Resume la \textbf{covarianza} de variables correlacionadas en pocos \textbf{componentes} (combinaciones lineales de las originales) \vspace{0.2cm}
\item El primero apunta en la dirección de \textbf{máxima varianza}; los demás son \textbf{ortogonales} y captan lo que queda \vspace{0.2cm}
\item Es \textbf{determinista}; los componentes resumen los datos pero \textbf{no} tienen un significado directo
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\includegraphics[width=\textwidth]{pca_idea.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCuántos componentes? (scree plot)}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Cada componente capta una fracción \textbf{decreciente} de la varianza \vspace{0.2cm}
\item El \textbf{scree plot} muestra ese aporte; nos quedamos en el ``codo'' \vspace{0.2cm}
\item Ejemplo (redes sociales, 36 palabras): caída fuerte tras el primero; con 5 componentes capturamos $\sim$27\% de la varianza
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{teens_scree.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Interpretar con los \textit{loadings}}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Los \textbf{loadings} (pesos) dicen cuánto aporta cada variable original a un componente \vspace{0.2cm}
\item Mirándolos le damos \textbf{sentido} a un componente, aunque PCA no lo etiquete \vspace{0.2cm}
\item El PC3 de los teens separa \textbf{religión} (jesus, bible, church) de \textbf{fiesta} (drunk, drugs, sex)
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{teens_pc3.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żPara qué usar PCA?}

\begin{itemize}
\item \textbf{Compresión} y \textbf{visualización} (graficar en 2--3 componentes) \vspace{0.3cm}

\item \textbf{Quitar multicolinealidad} antes de una regresión (los componentes no están correlacionados) \vspace{0.3cm}

\item \textbf{Preprocesamiento}: con pocas componentes, métodos como kNN o redes neuronales se vuelven factibles \vspace{0.3cm}

\item Costo: los componentes son \textbf{menos interpretables} que las variables originales

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Embeddings}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Un \textbf{embedding} representa un objeto complejo como un \textbf{vector denso} de pocos números \vspace{0.2cm}
\item \textbf{word2vec}: palabras como vectores; las de significado parecido quedan cerca \vspace{0.2cm}
\item Y las relaciones se vuelven \textbf{aritmética}: ``rey $-$ hombre $+$ mujer $\approx$ reina''
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{embeddings_analogy.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Autoencoders}

\begin{itemize}
\item Una red neuronal que aprende a \textbf{comprimir} los datos y luego \textbf{reconstruirlos} \vspace{0.3cm}

\item La capa central (un ``cuello de botella'') es una \textbf{representación compacta} aprendida \vspace{0.3cm}

\item Es la versión \textbf{profunda y no lineal} de la reducción de dimensión \vspace{0.3cm}

\item Las redes neuronales aprenden embeddings de palabras, imágenes, usuarios, productos... \vspace{0.3cm}

\item Es el puente hacia la última sesión: \textbf{aprendizaje profundo}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Resumen de la sesión}

\begin{itemize}
\item El aprendizaje \textbf{no supervisado} descubre estructura sin etiquetas \vspace{0.3cm}

\item \textbf{Clustering} (k-medias): agrupa objetos similares; elegir $k$ es clave \vspace{0.3cm}

\item \textbf{Detección de anomalías}: estadística, distancia, densidad (LOF) o modelos (isolation forest); el contexto define el umbral \vspace{0.3cm}

\item \textbf{Representación}: PCA resume muchas variables en pocas componentes; los \textit{embeddings} y \textit{autoencoders} extienden la idea hacia el aprendizaje profundo

\end{itemize}
}

\frame{
  \frametitle{Aplicación real en Colombia: UIAF}

\begin{itemize}
\item La \textbf{UIAF} (Unidad de Información y Análisis Financiero) es la unidad de \textbf{inteligencia financiera} del país \vspace{0.3cm}

\item Su misión: detectar \textbf{lavado de activos} y financiación del terrorismo en un océano de datos (miles de millones de registros) \vspace{0.3cm}

\item No hay una etiqueta de ``culpable/inocente'': hay que encontrar lo \textbf{atípico} y las \textbf{redes} sospechosas
\end{itemize}
}
