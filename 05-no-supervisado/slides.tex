%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Aprendizaje no supervisado}

\begin{itemize}
\item En el supervisado teníamos un \textit{outcome} $y$ que queríamos predecir \vspace{0.3cm}

\item En el \textbf{no supervisado} no hay $y$: solo tenemos las características \vspace{0.3cm}

\item El objetivo no es predecir, sino \textbf{describir} y \textbf{descubrir patrones} en los datos \vspace{0.3cm}

\item Ninguna variable es más importante que las demás: las miramos todas en conjunto

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Tres tareas que veremos}

\begin{itemize}
\item[1.] \textbf{Clustering}: agrupar objetos similares (segmentar) \vspace{0.3cm}

\item[2.] \textbf{Detección de anomalías}: encontrar los casos ``raros'' \vspace{0.3cm}

\item[3.] \textbf{Representación / reducción de dimensión}: resumir muchas variables en pocas, más útiles \vspace{0.3cm}

\item Como no hay etiquetas, evaluar el resultado es más difícil: mucho depende de la \textbf{interpretación} del analista

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żPara qué sirve?}

\begin{itemize}
\item \textbf{Segmentación} de clientes, votantes o mercados \vspace{0.3cm}

\item \textbf{Detección de fraude}, fallas o intrusiones (lo raro) \vspace{0.3cm}

\item \textbf{Visualización} y \textbf{compresión} de datos de alta dimensión \vspace{0.3cm}

\item \textbf{Preprocesamiento}: crear mejores variables para un modelo supervisado posterior

\end{itemize}
}

\frame{
  \frametitle{Clustering}

\begin{itemize}
\item A veces conviene \textbf{agrupar} objetos en categorías que no conocíamos de antemano \vspace{0.3cm}

\item En marketing: segmentar mercados para una estrategia por grupo \vspace{0.3cm}

\item En campańas políticas: identificar tipos de votantes \vspace{0.3cm}

\item El clustering divide los datos en \textbf{clusters}: grupos de objetos similares entre sí y distintos de los demás

\end{itemize}
}

\frame{
  \frametitle{Clustering: clasificación no supervisada}

\begin{itemize}
\item Es \textbf{no supervisado}: no decimos ex ante cómo deben verse los grupos \vspace{0.3cm}

\item No sirve para \textbf{predecir}, sino para \textbf{descubrir} estructura \vspace{0.3cm}

\item Los grupos que devuelve el computador \textbf{no tienen significado explícito}: la interpretación la pone el analista \vspace{0.3cm}

\item żCómo decide el computador dónde están las fronteras de los grupos?

\end{itemize}

\frame{
  \frametitle{El algoritmo de k-medias}

\begin{itemize}
\item El más usado para clustering; emparentado con kNN \vspace{0.3cm}

\item Asigna cada uno de los $n$ ejemplos a uno de $k$ clusters \vspace{0.3cm}

\item El número $k$ se fija \textbf{ex ante} (de forma heurística) \vspace{0.3cm}

\item Busca \textbf{minimizar} las diferencias dentro de cada cluster y \textbf{maximizar} las diferencias entre clusters

\end{itemize}
}

\frame{
  \frametitle{k-medias: elegir centros}

\begin{itemize}
\item Se eligen al azar $k$ puntos como centros iniciales (aquí $k=3$):
\end{itemize}

\includegraphics[width=0.6\textwidth]{clusters3.pdf}
\end{figure}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{k-medias: asignar al centro más cercano}

\begin{itemize}
\item Cada punto se asigna al centro más cercano (distancia euclídea): se forma el \textbf{diagrama de Voronoi}
\[
dist(x,y)=\sqrt{\sum_{i=1}^n (x_i-y_i)^2}
\]
\end{itemize}

\includegraphics[width=0.55\textwidth]{clusters4.pdf}
\end{figure}
}

\begin{itemize}
\item Cada centro se mueve al \textbf{centroide} (la posición promedio) de los puntos de su cluster:

\begin{itemize}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCuántos clusters?}

\begin{itemize}
\item $k$ es una decisión crucial (y el algoritmo también es sensible a los centros iniciales) \vspace{0.3cm}

\item Si hay un \textbf{prior}, úsalo (categorías de negocio, número de mesas, de campańas...) \vspace{0.3cm}

\item Regla de dedo: $k = \sqrt{n/2}$ \vspace{0.3cm}

\item \textbf{Método del codo}: aumentar $k$ hasta que la ganancia marginal en homogeneidad se aplane

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żQué es una anomalía?}

\begin{itemize}
\item Una \textbf{anomalía} (\textit{outlier}) es un caso que se aparta mucho del patrón general \vspace{0.3cm}

\item A menudo lo raro es lo más interesante:
\begin{itemize}
\item transacciones \textbf{fraudulentas} con tarjeta
\item \textbf{fallas} de una máquina o un sensor
\item \textbf{intrusiones} en una red; errores en los datos
\end{itemize}
\vspace{0.3cm}

\item Suele ser un problema \textbf{no supervisado}: no tenemos ejemplos etiquetados de ``anomalía''

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Cuatro familias de métodos}

\begin{itemize}
\item \textbf{Estadístico}: lo que cae lejos de lo típico (z-score, rango intercuartílico) \vspace{0.3cm}

\item \textbf{Por distancia}: puntos lejos de sus vecinos o del centro de su cluster \vspace{0.3cm}

\item \textbf{Por densidad}: puntos en zonas poco pobladas (LOF) \vspace{0.3cm}

\item \textbf{Por modelo}: aprender la ``forma normal'' y marcar lo que no encaja (\textit{isolation forest}, \textit{one-class SVM})

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Enfoque estadístico}

\begin{columns}
\begin{column}{0.5\textwidth}
\footnotesize
\begin{itemize}
\item Univariado: marcar valores con $|z|>3$ (regla de 3 sigma) o fuera del rango intercuartílico (Tukey) \vspace{0.2cm}
\item Multivariado: la \textbf{distancia de Mahalanobis} mide qué tan ``raro'' es un punto teniendo en cuenta la correlación entre variables \vspace{0.2cm}
\item Simple y rápido, pero asume una forma (p. ej. normal)
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{zscore_outliers.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Enfoque por distancia}

\begin{itemize}
\item Idea: un punto es anómalo si está \textbf{lejos} de los demás \vspace{0.3cm}

\item Distancia al $k$-ésimo vecino más cercano: a mayor distancia, más aislado \vspace{0.3cm}

\item O distancia al \textbf{centroide} de su cluster (conecta con k-medias): los puntos lejos de todos los centros son sospechosos \vspace{0.3cm}

\item Limitación: falla cuando hay zonas con densidades muy distintas

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LOF: Local Outlier Factor}

\begin{itemize}
\item Idea: comparar la \textbf{densidad local} de un punto con la de sus vecinos \vspace{0.3cm}

\item Si un punto está en una zona \textbf{mucho menos densa} que sus vecinos, es sospechoso \vspace{0.3cm}

\item Devuelve un \textbf{puntaje}:
\begin{itemize}
\item $LOF \approx 1$: densidad parecida a la de sus vecinos (normal)
\item $LOF \gg 1$: mucho más aislado que sus vecinos (anomalía)
\end{itemize}
\vspace{0.3cm}

\item Ventaja sobre el enfoque por distancia: detecta anomalías \textbf{locales}, no solo las globales

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LOF: żcómo se calcula? (intuición)}

\begin{itemize}
\item[1.] Para cada punto, se miran sus $k$ vecinos más cercanos \vspace{0.3cm}

\item[2.] Se estima su \textbf{densidad local}: qué tan apińados están esos vecinos a su alrededor \vspace{0.3cm}

\item[3.] Se compara esa densidad con la densidad \textbf{de} sus vecinos \vspace{0.3cm}

\item[4.] El \textbf{LOF} es la razón entre ambas: si la del punto es mucho menor, el cociente es alto \vspace{0.3cm}

\item Por eso un punto solitario al borde de un grupo denso tiene LOF alto, aunque no esté ``tan lejos'' en distancia absoluta

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{LOF en acción}

\begin{figure}
\includegraphics[width=0.6\textwidth]{lof_anomalias.pdf}
\end{figure}

\begin{center}
\small Los puntos con LOF alto (rojos) están en zonas poco densas, lejos de los grupos.
\end{center}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Isolation Forest}

\begin{itemize}
\item Un enfoque \textbf{por modelo}, basado en árboles (conecta con random forests) \vspace{0.3cm}

\item Idea: aislar cada punto haciendo \textbf{cortes aleatorios} en las variables \vspace{0.3cm}

\item Un punto \textbf{anómalo}, al estar apartado, se aísla con \textbf{muy pocos} cortes \vspace{0.3cm}

\item Uno \textbf{normal}, en una zona densa, requiere muchos cortes \vspace{0.3cm}

\item El número promedio de cortes mide la anomalía. Es rápido y escala bien a muchos datos

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCómo evaluar sin etiquetas?}

\begin{itemize}
\item Sin etiquetas de ``anomalía'', validar es difícil: a menudo se \textbf{inspeccionan a mano} los casos con mayor puntaje \vspace{0.3cm}

\item Si tenemos \textbf{algunas} etiquetas (p. ej. fraudes confirmados), es un problema \textbf{semi-supervisado}: podemos medir cuántos de los \textit{top-N} marcados eran reales (precisión@N) \vspace{0.3cm}

\item No hay umbral universal: cuántos casos marcar depende del \textbf{costo} de cada error \vspace{0.3cm}

\item Como se basa en distancias, conviene \textbf{escalar} las variables (como en kNN)

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{El reto de la alta dimensión}

\begin{itemize}
\item Muchos problemas tienen \textbf{decenas o miles} de variables \vspace{0.3cm}

\item Difíciles de \textbf{visualizar}, costosas de procesar, y con la \textbf{maldición de la dimensionalidad} \vspace{0.3cm}

\item Pero muchas variables están \textbf{correlacionadas}: hay redundancia \vspace{0.3cm}

\item żPodemos \textbf{resumir} la información en muchas menos dimensiones, perdiendo poco?

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Aprendizaje de representación}

\begin{itemize}
\item En vez de usar las variables crudas, buscamos una \textbf{representación} más compacta y útil de los datos \vspace{0.3cm}

\item Es la versión no supervisada del \textit{feature extraction} que vimos en desempeńo \vspace{0.3cm}

\item La técnica clásica y más intuitiva: \textbf{Análisis de Componentes Principales} (PCA) \vspace{0.3cm}

\item (Y la base conceptual de los \textit{embeddings} del aprendizaje profundo)

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{PCA: la idea}

\begin{columns}
\begin{column}{0.52\textwidth}
\begin{itemize}
\item PCA encuentra nuevas variables (\textbf{componentes}) que son combinaciones lineales de las originales \vspace{0.2cm}
\item La 1.Ş componente apunta en la dirección de \textbf{máxima varianza} \vspace{0.2cm}
\item Las siguientes son ortogonales y captan la varianza restante
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\includegraphics[width=\textwidth]{pca_idea.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{PCA: żcómo funciona?}

\begin{itemize}
\item Buscamos direcciones que maximicen la \textbf{varianza} de los datos proyectados \vspace{0.3cm}

\item Esas direcciones son los \textbf{autovectores} de la matriz de covarianzas; cada \textbf{autovalor} dice cuánta varianza captura su dirección \vspace{0.3cm}

\item Las componentes son \textbf{ortogonales} (no correlacionadas) y se ordenan de mayor a menor varianza \vspace{0.3cm}

\item Intuición, sin fórmulas: ``girar los ejes'' para alinearlos con la mayor dispersión de la nube de puntos

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{PCA: de 30 variables a 2}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Tomamos las 30 variables del cáncer de seno (sesión 1) \vspace{0.2cm}
\item Las proyectamos en sus 2 primeras componentes \vspace{0.2cm}
\item Aunque PCA \textbf{no} usa el diagnóstico, los benignos y malignos quedan casi separados
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{pca_projection.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCuántas componentes?}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Cada componente explica una fracción de la \textbf{varianza} total \vspace{0.2cm}
\item Aquí, las 2 primeras ya capturan $\sim$63\% \vspace{0.2cm}
\item Nos quedamos con las que acumulan ``suficiente'' (otra vez, el codo)
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{pca_scree.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Interpretar las componentes}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Cada componente es una \textbf{mezcla} de las variables originales \vspace{0.2cm}
\item Los \textbf{pesos} (\textit{loadings}) dicen cuánto aporta cada variable \vspace{0.2cm}
\item Mirándolos le damos sentido a la componente (aquí, PC1 $\approx$ ``tamańo/severidad'' del tumor)
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{pca_loadings.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żPara qué usar PCA?}

\begin{itemize}
\item \textbf{Compresión}: menos columnas, menos cómputo y almacenamiento \vspace{0.3cm}

\item \textbf{Visualización}: graficar en 2 o 3 componentes \vspace{0.3cm}

\item \textbf{Quitar multicolinealidad} antes de una regresión (las componentes no están correlacionadas) \vspace{0.3cm}

\item \textbf{Preprocesamiento} para otros modelos \vspace{0.3cm}

\item Costo: las componentes son \textbf{menos interpretables} que las variables originales

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Más allá de PCA: t-SNE y UMAP}

\begin{itemize}
\item PCA es \textbf{lineal}. Para estructuras curvas hay métodos no lineales \vspace{0.3cm}

\item \textbf{t-SNE} y \textbf{UMAP}: pensados sobre todo para \textbf{visualizar} datos de alta dimensión en 2D \vspace{0.3cm}

\item Preservan la \textbf{vecindad local}: puntos parecidos quedan cerca \vspace{0.3cm}

\item Muy usados para explorar imágenes, texto o datos genéticos

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{t-SNE/UMAP: cómo leerlos}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Lo \textbf{cercano} es similar; los grupos saltan a la vista \vspace{0.2cm}
\item \textbf{Cuidado}: NO conservan distancias globales ni tamańos de cluster \vspace{0.2cm}
\item No interpretes el tamańo de los grupos ni qué tan separados están como algo literal
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{tsne_concept.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Embeddings}

\begin{columns}
\begin{column}{0.5\textwidth}
\begin{itemize}
\item Un \textbf{embedding} representa un objeto complejo como un \textbf{vector denso} de pocos números \vspace{0.2cm}
\item \textbf{word2vec}: palabras como vectores; las de significado parecido quedan cerca \vspace{0.2cm}
\item Y las relaciones se vuelven \textbf{aritmética}: ``rey $-$ hombre $+$ mujer $\approx$ reina''
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\includegraphics[width=\textwidth]{embeddings_analogy.pdf}
\end{column}
\end{columns}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{itemize}
\item Una red neuronal que aprende a \textbf{comprimir} los datos y luego \textbf{reconstruirlos} \vspace{0.3cm}

\item La capa central (un ``cuello de botella'') es una \textbf{representación compacta} aprendida \vspace{0.3cm}

\item Es la versión \textbf{profunda y no lineal} de la reducción de dimensión \vspace{0.3cm}

\item Las redes neuronales aprenden embeddings de palabras, imágenes, usuarios, productos... \vspace{0.3cm}

\item Es el puente hacia la última sesión: \textbf{aprendizaje profundo}

\end{itemize}
