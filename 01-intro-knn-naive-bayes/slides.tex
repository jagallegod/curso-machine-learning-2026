\frame{
  \frametitle{Un principio básico}

\begin{itemize}

\item Para clasificar partimos de una idea muy intuitiva \vspace{0.3cm}

\item \textbf{Las cosas que son parecidas tienen propiedades parecidas} \vspace{0.3cm}

\item ``Blanco es, gallina lo pone, frito se come'' \vspace{0.3cm}

\item Los algoritmos de \textit{machine learning} usan este principio para clasificar \vspace{0.3cm}

\item En una misma categoría caen los elementos similares, o \textit{vecinos más cercanos}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{El algoritmo kNN}

\begin{itemize}

\item El más popular es \textbf{kNN} (\textit{k nearest neighbors}, k vecinos más cercanos) \vspace{0.3cm}

\item Ventajas:
\begin{itemize}
\item Simple y efectivo
\item No hace supuestos sobre la distribución de los datos
\item Fase de entrenamiento muy rápida
\end{itemize}
\vspace{0.2cm}

\item Desventajas:
\begin{itemize}
\item No produce un modelo: es difícil ver la relación entre variables
\item Requiere elegir bien el valor de k
\item Fase de clasificación lenta
\end{itemize}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żCómo clasifica kNN?}

\begin{itemize}

\item Partimos de un conjunto de datos ya clasificados (entrenamiento) \vspace{0.3cm}

\item Elegimos un número k de vecinos \vspace{0.3cm}

\item Para cada caso \textbf{sin clasificar}, buscamos sus k vecinos más cercanos \vspace{0.3cm}

\item El caso se asigna a la categoría de la \textbf{mayoría} de esos vecinos

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Ejemplo: clasificar alimentos}

\begin{itemize}

\item A una persona con los ojos vendados se le pide clasificar un alimento \vspace{0.3cm}

\item Tenemos una base de otros alimentos, descritos por dos características: \vspace{0.3cm}

\begin{itemize}
\item Qué tan crujiente es (de 1 a 10)
\item Qué tan dulce es (de 1 a 10)
\end{itemize}
\vspace{0.3cm}

\item Y sabemos su grupo: fruta, vegetal o proteína

\end{itemize}
}

\frame{
  \frametitle{Distancia euclídea}

\begin{itemize}

\item Para hallar los vecinos más cercanos necesitamos una medida de distancia \vspace{0.3cm}

\item La más usada es la \textbf{distancia euclídea}: la longitud del segmento que une dos puntos \vspace{0.3cm}

\item Entre dos vectores $m$-dimensionales, $p$ y $q$:

\[
dist(p,q)=\sqrt{\sum_{i=1}^m (p_i-q_i)^2}
\]

\end{itemize}
}

\frame{
  \frametitle{Distancia al tomate}

\begin{itemize}
\item Calculamos la distancia del tomate a cada alimento:
\end{itemize}

\begin{figure}
\includegraphics[scale=0.65]{distance.pdf}
\end{figure}

\begin{itemize}
\item Con $k=1$, el vecino más cercano es la naranja: sería \textbf{fruta} \vspace{0.2cm}
\item Con $k=3$ (naranja, uva, nueces): también \textbf{fruta}
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{La elección de k}

\begin{itemize}

\item Cuántos vecinos usar es una decisión crucial: hay un \textit{trade-off} sesgo--varianza \vspace{0.3cm}

\item k \textbf{grande}: menos sensible al ruido, pero puede ignorar patrones pequeńos \vspace{0.3cm}

\item k \textbf{pequeńo}: capta detalles, pero una observación ruidosa puede dańar la clasificación \vspace{0.3cm}

\item No hay regla universal. Una heurística común es $k=\sqrt{n}$ \vspace{0.3cm}

\item Mejor aún: probar varios k y quedarse con el que clasifique mejor

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Preparación de los datos}

\begin{itemize}

\item Para calcular distancias, todas las características deben estar en la \textbf{misma escala} \vspace{0.3cm}

\item Si no, una variable con valores grandes dominaría solo por su escala \vspace{0.3cm}

\item Dos formas de rescalar:

\end{itemize}

\[
\text{Normalización min-max:}\quad X' = \frac{X - X_{min}}{X_{max}-X_{min}}
\]

\[
\text{Estandarización (z-score):}\quad X' = \frac{X - \mu}{\sigma}
\]
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Preparación de los datos}

\begin{itemize}

\item La \textbf{normalización min-max} deja todos los valores entre 0 y 1 \vspace{0.3cm}

\item La \textbf{estandarización} mide cuántas desviaciones estándar se aleja un valor de la media \vspace{0.3cm}

\item żY las variables nominales? La distancia euclídea no está definida para ellas \vspace{0.3cm}

\item Solución: crear variables \textit{dummy} (si hay $n$ categorías, se crean $n-1$ dummies)

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{kNN: aprendizaje perezoso}

\begin{itemize}

\item A kNN se le llama un algoritmo \textbf{perezoso} (\textit{lazy}) \vspace{0.3cm}

\item No hay abstracción: no se construye un modelo, solo se almacenan los datos \vspace{0.3cm}

\item Es una técnica \textbf{no paramétrica}: no supone una forma funcional \vspace{0.3cm}

\item Ventaja: flexible. Desventaja: no entendemos cómo se relacionan las variables con el \textit{outcome} \vspace{0.3cm}

\item \textbf{Al taller}: diagnóstico de cáncer de seno con la base \texttt{wisc\_bc\_data.csv}

\end{itemize}

\frame{
  \frametitle{Pronósticos probabilísticos}

\begin{itemize}

\item A veces la economía se parece más a la meteorología que a la física \vspace{0.3cm}

\item Un meteorólogo no dice ``va a llover''; dice ``hay 70\% de probabilidad de lluvia'' \vspace{0.3cm}

\item Basa su pronóstico en eventos pasados \vspace{0.3cm}

\item El método de hoy hace justamente eso: pronostica de manera \textbf{probabilística} \vspace{0.3cm}

\item Usa datos del pasado para anticipar eventos cuyo estado desconocemos

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Naive Bayes}

\begin{itemize}

\item El algoritmo \textbf{Naive Bayes} se basa en el \textbf{teorema de Bayes} \vspace{0.3cm}

\item Usa datos de entrenamiento para calcular la probabilidad de un resultado \vspace{0.3cm}

\item Esa probabilidad depende de las características del objeto a clasificar \vspace{0.3cm}

\item Se aplica mucho en:
\begin{itemize}
\item Clasificación de textos (filtros de spam)
\item Detección de anomalías o intrusiones en redes
\item Diagnóstico médico a partir de síntomas
\end{itemize}

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Un poco de probabilidad}

\begin{itemize}

\item Sea $P(A)$ la probabilidad de que ocurra el evento $A$ \vspace{0.3cm}

\item Su complemento $\neg A$ es la no ocurrencia de $A$, con $P(\neg A)=1-P(A)$ \vspace{0.3cm}

\item Dos eventos son \textbf{independientes} si la ocurrencia de uno no informa sobre el otro: $P(A\cap B)=P(A)\cdot P(B)$ \vspace{0.3cm}

\item Pero los casos interesantes son los \textbf{dependientes}: ahí un evento sí nos dice algo del otro

\end{itemize}
}

\frame{
  \frametitle{Spam y la palabra ``Viagra''}

żQué nos dice sobre la probabilidad de spam el que un correo contenga ``Viagra''?

\begin{figure}
\includegraphics[scale=0.9]{viagra.pdf}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{El teorema de Bayes}

\begin{itemize}

\item La probabilidad condicional de $A$ dado $B$ es:

\[
P(A\mid B) = \frac{P(A\cap B)}{P(B)}
\]

\item Reorganizando ($P(A\cap B)=P(B\mid A)\cdot P(A)$), llegamos a la forma más útil:

\[
P(A\mid B) = \frac{P(B\mid A)\cdot P(A)}{P(B)}
\]

\item Nos permite ``actualizar nuestra creencia'' sobre $A$ al observar $B$

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Ejemplo: spam y Viagra}

\[
P(\text{spam}\mid \text{Viagra})=\frac{P(\text{Viagra}\mid \text{spam})\cdot P(\text{spam})}{P(\text{Viagra})}
\]

\vspace{0.3cm}

\begin{itemize}
\item $P(\text{spam}\mid \text{Viagra})$: probabilidad \textit{posterior} \vspace{0.2cm}
\item $P(\text{spam})$: probabilidad \textit{a priori} \vspace{0.2cm}
\item $P(\text{Viagra}\mid \text{spam})$: \textit{verosimilitud} (likelihood) \vspace{0.2cm}
\item $P(\text{Viagra})$: verosimilitud marginal
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Ejemplo: spam y Viagra}

Con una tabla de frecuencias de 100 correos:

\begin{figure}
\includegraphics[scale=0.7]{freqtable.pdf}
\end{figure}

\begin{itemize}
\item $P(\text{spam})=20/100=0.2$;\; $P(\text{Viagra}\mid \text{spam})=4/20=0.2$;\; $P(\text{Viagra})=5/100=0.05$
\item Entonces $P(\text{spam}\mid \text{Viagra})=\dfrac{0.2\cdot 0.2}{0.05}=0.8$
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{żPor qué ``naive'' (ingenuo)?}

\begin{itemize}

\item Un filtro real usa muchas palabras, no una sola \vspace{0.3cm}

\item Naive Bayes hace dos supuestos ``ingenuos'' sobre las características: \vspace{0.3cm}

\begin{itemize}
\item[1.] Que todas son \textbf{igual de importantes}
\item[2.] Que son \textbf{independientes} entre sí
\end{itemize}
\vspace{0.3cm}

\item Sabemos que rara vez es cierto (``Viagra'', ``droga'' y ``prescripción'' suelen ir juntas) \vspace{0.3cm}

\item Pero aun así el método \textbf{funciona muy bien} y suele ser el primer candidato

\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{Varias características a la vez}

\begin{itemize}
\item El supuesto de independencia simplifica enormemente el cálculo \vspace{0.3cm}
\item Con cuatro palabras $W_1,\ldots,W_4$, la verosimilitud conjunta se factoriza:
\end{itemize}

\footnotesize
\begin{align*}
&P(\text{spam}\mid W_1\cap \neg W_2 \cap \neg W_3 \cap W_4)\;\propto\\
&\quad P(W_1\mid \text{spam})\cdot P(\neg W_2\mid \text{spam}) \cdot P(\neg W_3\mid \text{spam}) \cdot P(W_4\mid \text{spam}) \cdot P(\text{spam})
\end{align*}
\normalsize

\begin{itemize}
\item Cada término sale de la tabla de frecuencias: un producto de proporciones simples
\end{itemize}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{
  \frametitle{El estimador de Laplace}

\begin{itemize}

\item żQué pasa si una palabra aparece \textbf{0 veces} en una clase? \vspace{0.3cm}

\item Todo el producto se vuelve 0, y la probabilidad posterior también \vspace{0.3cm}

\item Eso no tiene sentido si las demás palabras sí apuntan a spam \vspace{0.3cm}

\item \textbf{Solución (Laplace)}: sumar 1 a cada conteo de la tabla de frecuencias \vspace{0.3cm}

\item Así evitamos los ceros y el método se vuelve más robusto

\end{itemize}
}

\frame{
  \frametitle{Características numéricas}

\begin{itemize}

\item Hasta aquí las características eran dicotómicas (la palabra está o no está) \vspace{0.3cm}

\item żY si una característica es numérica? Por ejemplo, la hora de envío del correo \vspace{0.3cm}

\item Estrategia: \textbf{discretizar} la variable en categorías razonables \vspace{0.3cm}

\item Por ejemplo, dividir el día en franjas (madrugada, mańana, tarde, noche) \vspace{0.3cm}

\item \textbf{Al taller}: predecir el partido de legisladores de EE.UU. con la base \texttt{votos84.csv}

\end{itemize}
